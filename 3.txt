--// FRONTMOST NOTIFICATION WITH EXECUTOR NAME + PID //--

local TweenService = game:GetService("TweenService")
local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer

-- GET EXECUTOR NAME
local executorName = "Unknown Executor"
if type(identifyexecutor) == "function" then
    executorName = identifyexecutor()
elseif type(syn) == "table" then
    executorName = "Synapse X"
elseif type(Krnl) == "table" then
    executorName = "KRNL"
elseif type(Fluxus) == "table" then
    executorName = "Fluxus"
elseif type(ProtoSmasher) == "table" then
    executorName = "ProtoSmasher"
end

-- GENERATE 6-DIGIT RANDOM PID
local pid = ""
for i = 1, 6 do
    pid = pid .. tostring(math.random(0,9))
end

-- FULL MESSAGE
local messageText = executorName .. " loaded successfully into the game | PID: " .. pid

-- CHOOSE GUI PARENT: CoreGui if accessible
local guiParent = game:GetService("CoreGui")
local success, _ = pcall(function()
    local test = Instance.new("ScreenGui")
    test.Parent = guiParent
    test:Destroy()
end)
if not success then
    guiParent = LocalPlayer:WaitForChild("PlayerGui")
end

-- REMOVE OLD GUI
if guiParent:FindFirstChild("CustomNotify") then
    guiParent.CustomNotify:Destroy()
end

-- CREATE SCREEN GUI
local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Name = "CustomNotify"
ScreenGui.ResetOnSpawn = false
ScreenGui.IgnoreGuiInset = true
ScreenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling
ScreenGui.Parent = guiParent

-- NOTIFICATION FRAME
local Frame = Instance.new("Frame")
Frame.Parent = ScreenGui
Frame.AnchorPoint = Vector2.new(0.5, 1)
Frame.Position = UDim2.new(0.5, 0, 0.98, 0)
Frame.Size = UDim2.new(0, 0, 0, 30)
Frame.BackgroundColor3 = Color3.fromRGB(20, 20, 20)
Frame.BackgroundTransparency = 0.15
Frame.BorderSizePixel = 0
Frame.ClipsDescendants = true
Frame.ZIndex = 9999
local UICorner = Instance.new("UICorner", Frame)
UICorner.CornerRadius = UDim.new(0, 15)

-- GREEN CHECK
local Tick = Instance.new("TextLabel")
Tick.Parent = Frame
Tick.BackgroundTransparency = 1
Tick.Size = UDim2.new(0, 24, 1, 0)
Tick.Position = UDim2.new(1, -35, 0, 0)
Tick.Font = Enum.Font.GothamSemibold
Tick.Text = "âœ“"
Tick.TextColor3 = Color3.fromRGB(0, 255, 0)
Tick.TextScaled = false
Tick.TextSize = 18
Tick.ZIndex = 10000

-- MESSAGE TEXT
local Message = Instance.new("TextLabel")
Message.Parent = Frame
Message.BackgroundTransparency = 1
Message.AnchorPoint = Vector2.new(0.5, 0)
Message.Position = UDim2.new(0.5, -20, 0, 0)
Message.Size = UDim2.new(1, -70, 1, 0)
Message.Font = Enum.Font.GothamSemibold
Message.Text = messageText
Message.TextColor3 = Color3.fromRGB(255, 255, 255)
Message.TextScaled = false
Message.TextSize = 16
Message.TextWrapped = true
Message.ZIndex = 10000

-- OPEN ANIMATION (EXPAND)
TweenService:Create(Frame, TweenInfo.new(0.5, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {
    Size = UDim2.new(0.7, 0, 0, 30)
}):Play()

task.wait(2)

-- CLOSE ANIMATION (SHRINK)
local shrink = TweenService:Create(Frame, TweenInfo.new(0.4, Enum.EasingStyle.Quint, Enum.EasingDirection.In), {
    Size = UDim2.new(0, 0, 0, 30)
})
shrink:Play()
shrink.Completed:Wait()

-- FINAL FADE & CIRCLE
UICorner.CornerRadius = UDim.new(0, 100)
TweenService:Create(Frame, TweenInfo.new(0.3, Enum.EasingStyle.Quad), {
    BackgroundTransparency = 1
}):Play()

task.wait(0.3)
ScreenGui:Destroy()
